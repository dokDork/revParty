function Write-OutputConditional {
    param(
        [string]$Message,
        [string]$ForegroundColor = "White",
        [string]$BackgroundColor = $null
    )
    
    if (-not $SILENT) {
        if ($BackgroundColor) {
            Write-Host $Message -ForegroundColor $ForegroundColor -BackgroundColor $BackgroundColor
        } else {
            Write-Host $Message -ForegroundColor $ForegroundColor
        }
    }
}

function Test-SystemVirtualization {
    # Check if running under Hyper-V (common on Windows 11)
    try {
        $hyperv = Get-WmiObject -Class Win32_ComputerSystem -ErrorAction SilentlyContinue
        if ($hyperv -and $hyperv.HypervisorPresent -eq $true) {
            if ($Verbose -and -not $SILENT) { 
                Write-OutputConditional "[INFO] Hyper-V detected - system virtualization present" -ForegroundColor Cyan 
            }
            return $true
        }
    } catch {}
    
    # Check registry for Hyper-V
    try {
        $regKey = Get-ItemProperty -Path "HKLM:\SOFTWARE\Microsoft\Windows NT\CurrentVersion\Virtualization" -ErrorAction SilentlyContinue
        if ($regKey -and $regKey.InstallDate) {
            if ($Verbose -and -not $SILENT) { 
                Write-OutputConditional "[INFO] System virtualization registry detected" -ForegroundColor Cyan 
            }
            return $true
        }
    } catch {}
    
    return $false
}

function Test-SandboxEnvironment {
    param([bool]$SystemVirtualization)
    
    $sandboxDetected = $false
    $detectionResults = @()
    
    # Technique 1: Check for sandbox-specific processes
    $sandboxProcesses = @(
        "sandboxie.exe", "sandboxierpc.exe", 
        "procmon.exe", "wireshark.exe", "tcpdump.exe", "fiddler.exe", "api-mon.exe"
    )
    
    $runningProcesses = Get-Process -ErrorAction SilentlyContinue | 
        Select-Object -ExpandProperty Name -Unique
    
    foreach ($process in $sandboxProcesses) {
        if ($runningProcesses -contains $process) {
            $detectionResults += "Sandbox analysis tool detected: $process"
            $sandboxDetected = $true
        }
    }
    
    # Technique 2: Check CPU for virtualization indicators
    $cpuInfo = Get-WmiObject -Class Win32_Processor -ErrorAction SilentlyContinue
    if ($cpuInfo) {
        $cpuName = $cpuInfo.Name
        
        # Only flag if NOT running under system virtualization
        if (-not $SystemVirtualization) {
            if ($cpuName -match "QEMU|KVM|Xen|Parallels") {
                $detectionResults += "Emulated CPU detected: $cpuName"
                $sandboxDetected = $true
            }
        }
        
        # Check for very low core count (common in sandboxes)
        if ($cpuInfo.NumberOfCores -lt 2) {
            $detectionResults += "Very low core count: $($cpuInfo.NumberOfCores) cores"
            $sandboxDetected = $true
        }
    }
    
    # Technique 3: Check system memory
    $memoryInfo = Get-WmiObject -Class Win32_ComputerSystem -ErrorAction SilentlyContinue
    if ($memoryInfo) {
        $totalMemoryGB = [math]::Round($memoryInfo.TotalPhysicalMemory / 1GB, 2)
        if ($totalMemoryGB -lt $MemoryThresholdGB) {
            $detectionResults += "Low memory: ${totalMemoryGB}GB (threshold: ${MemoryThresholdGB}GB)"
            $sandboxDetected = $true
        }
    }
    
    # Technique 4: Check disk size
    $disks = Get-WmiObject -Class Win32_LogicalDisk -ErrorAction SilentlyContinue
    foreach ($disk in $disks) {
        if ($disk.DeviceID -eq "C:") {
            $totalSizeGB = [math]::Round($disk.Size / 1GB, 2)
            if ($totalSizeGB -lt $DiskThresholdGB) {
                $detectionResults += "Small disk: ${totalSizeGB}GB (threshold: ${DiskThresholdGB}GB)"
                $sandboxDetected = $true
            }
        }
    }
    
    # Technique 5: Check MAC addresses with system virtualization awareness
    if (-not $SystemVirtualization) {
        $networkAdapters = Get-WmiObject -Class Win32_NetworkAdapter -ErrorAction SilentlyContinue
        foreach ($adapter in $networkAdapters) {
            if ($adapter.MACAddress -and $adapter.NetEnabled -eq $true) {
                $macPrefix = $adapter.MACAddress.Substring(0, 8).Replace(":", "").ToUpper()
                $virtualPrefixes = @("000C29", "001C14", "005056", "080027", "0A0027")
                
                if ($virtualPrefixes -contains $macPrefix) {
                    $adapterName = $adapter.Name
                    if ($adapterName -notmatch "Hyper-V|Virtual|WSL|Docker|vEthernet") {
                        $detectionResults += "Suspicious MAC from non-system adapter: $($adapter.MACAddress) ($adapterName)"
                        $sandboxDetected = $true
                    }
                }
            }
        }
    }
    
    # Technique 6: Check uptime
    try {
        $os = Get-CimInstance -ClassName Win32_OperatingSystem -ErrorAction SilentlyContinue
        if ($os) {
            $uptime = (Get-Date) - $os.LastBootUpTime
            if ($uptime.TotalMinutes -lt $UptimeThresholdMinutes) {
                $detectionResults += "Very short uptime: $($uptime.TotalMinutes) minutes (threshold: ${UptimeThresholdMinutes} minutes)"
                $sandboxDetected = $true
            }
        }
    } catch {}
    
    # Technique 7: Check screen resolution
    try {
        Add-Type -AssemblyName System.Windows.Forms
        $screenWidth = [System.Windows.Forms.SystemInformation]::PrimaryMonitorSize.Width
        $screenHeight = [System.Windows.Forms.SystemInformation]::PrimaryMonitorSize.Height
        if ($screenWidth -lt 1024 -or $screenHeight -lt 768) {
            $detectionResults += "Low screen resolution: ${screenWidth}x${screenHeight}"
            $sandboxDetected = $true
        }
    } catch {
        # Skip if can't check resolution
    }
    
    return @{
        Detected = $sandboxDetected
        Results = $detectionResults
    }
}

# -- Funzioni helper --------------------------------------------

function out($msg, $color = "White") {
    if (-not $SILENT) {
        if ($color -eq "White") { Write-Host $msg }
        else { Write-Host $msg -ForegroundColor $color }
    }
}

function dbg($msg) {
    if ($DEBUG -and -not $SILENT) {
        $ts = (Get-Date).ToString("HH:mm:ss.fff")
        Write-Host "  [DBG $ts] $msg" -ForegroundColor Cyan
    }
}

function Send-Command($stream, $cmd) {
    dbg ">>> Invio comando: $cmd"
    $bytes = [System.Text.Encoding]::ASCII.GetBytes($cmd + "`r`n")
    try {
        $stream.Write($bytes, 0, $bytes.Length)
        $stream.Flush()
        dbg "    Comando inviato ($($bytes.Length) bytes)"
    } catch {
        out "[ERRORE] Impossibile inviare il comando: $_" "Red"
        throw
    }
    out "[C->S] $cmd"
}

function Read-Response($reader) {
    dbg "    In attesa risposta dal server (timeout $($READ_TIMEOUT_MS)ms)..."
    try {
        $line = $reader.ReadLine()
        if ($null -eq $line) {
            out "[ERRORE] Connessione chiusa dal server (ReadLine = null)" "Red"
            throw "Server ha chiuso la connessione"
        }
        dbg "    Risposta ricevuta: $line"
        out "[S->C] $line"
        return $line
    } catch {
        out "[ERRORE] Timeout o errore lettura risposta: $_" "Red"
        throw
    }
}

function New-TcpConnection($ip, $port) {
    dbg "Tentativo connessione TCP a ${ip}:${port} (timeout $($CONNECT_TIMEOUT_MS)ms) ..."
    $client = New-Object System.Net.Sockets.TcpClient
    $ar = $client.BeginConnect($ip, $port, $null, $null)
    $ok = $ar.AsyncWaitHandle.WaitOne($CONNECT_TIMEOUT_MS, $false)
    if (-not $ok) {
        $client.Close()
        throw "Timeout connessione TCP a ${ip}:${port} dopo $($CONNECT_TIMEOUT_MS)ms"
    }
    try { $client.EndConnect($ar) } catch {
        $client.Close()
        throw "Connessione TCP a ${ip}:${port} fallita: $_"
    }
    dbg "Connessione TCP stabilita. LocalEndPoint=$($client.Client.LocalEndPoint)"
    return $client
}
